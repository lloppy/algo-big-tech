class MaxStack {
    private val mainList = mutableListOf<Int>()
    private val maxList = mutableListOf<Int>()

    fun push(x: Int) {
        mainList.add(x)
        if (maxList.isEmpty() || x >= maxList.last()) {
            maxList.add(x)
        } else {
            maxList.add(maxList.last())
        }
    }

    fun pop(): Int {
        maxList.removeLast()
        return mainList.removeLast()
    }

    fun top(): Int {
        return mainList.last()
    }

    fun peekMax(): Int {
        return maxList.last()
    }

    fun popMax(): Int {
        val max = peekMax()
        val tempList = mutableListOf<Int>()

        // –ù–∞—Ö–æ–¥–∏–º –∏ —É–¥–∞–ª—è–µ–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç
        while (top() != max) {

        }
        val result = pop() // —É–¥–∞–ª—è–µ–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç

            push(tempList[i])
        }

        return result
    }

    fun isEmpty(): Boolean {
        return mainList.isEmpty()
    }
}
