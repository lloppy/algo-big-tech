```kotlin
class Solution {
    fun numSquares(n: Int): Int {
        val result = IntArray(n + 1) {Int.MAX_VALUE}
        result[0] = 0 // –î–ª—è —á–∏—Å–ª–∞ 0 –Ω—É–∂–Ω–æ 0 –∫–≤–∞–¥—Ä–∞—Ç–æ–≤

        // –î–ª—è –∫–∞–∂–¥–æ–≥–æ —á–∏—Å–ª–∞ i –≤—ã—á–∏—Å–ª—è–µ–º MAX –∫–≤–∞–¥—Ä–∞—Ç, –∫–æ—Ç–æ—Ä—ã–π –º–æ–∂–µ—Ç –≤ –Ω–µ–≥–æ –ø–æ–º–µ—Å—Ç–∏—Ç—å—Å—è
        for (i in 1..n) {

            val maxJ = sqrt(i.toDouble()).toInt() 
            for (j in maxJ downTo 1) {
                // result[i - j * j] - –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–≤–∞–¥—Ä–∞—Ç–æ–≤ –¥–ª—è –æ—Å—Ç–∞—Ç–∫–∞ 
                // –ø–ª—é—Å —Ç–µ–∫—É—â–∏–π –∫–≤–∞–¥—Ä–∞—Ç
                result[i] = minOf(result[i], result[i - j * j] + 1)
            }
        }
        return result[n] // –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–≤–∞–¥—Ä–∞—Ç–æ–≤ –¥–ª—è —á–∏—Å–ª–∞ n
    }
}
```
